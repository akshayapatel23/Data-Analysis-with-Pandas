import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset
data = pd.read_csv('sample_sales_data.csv')

# Display the first few rows of the dataset
print("First few rows of the dataset:")
print(data.head())

# Data Exploration
# Filter data (e.g., sales greater than $500)
filtered_data = data[data['Sales'] > 500]
print("\nFiltered data (Sales > 500):")
print(filtered_data)

# Sort data by Sales
sorted_data = data.sort_values(by='Sales', ascending=False)
print("\nData sorted by Sales:")
print(sorted_data.head())

# Group data by Product and calculate total sales for each product
grouped_data = data.groupby('Product')['Sales'].sum().reset_index()
print("\nTotal sales for each product:")
print(grouped_data)

# Summary Statistics
mean_sales = data['Sales'].mean()
median_sales = data['Sales'].median()
std_sales = data['Sales'].std()

print("\nSummary Statistics:")
print(f"Mean Sales: {mean_sales}")
print(f"Median Sales: {median_sales}")
print(f"Standard Deviation of Sales: {std_sales}")

# Data Visualization
# Histogram of Sales
plt.figure(figsize=(10, 6))
sns.histplot(data['Sales'], kde=True)
plt.title('Distribution of Sales')
plt.xlabel('Sales')
plt.ylabel('Frequency')
plt.show()

# Boxplot of Sales by Product
plt.figure(figsize=(10, 6))
sns.boxplot(x='Product', y='Sales', data=data)
plt.title('Sales by Product')
plt.xlabel('Product')
plt.ylabel('Sales')
plt.show()

# Scatter plot of Sales vs. Quantity
plt.figure(figsize=(10, 6))
sns.scatterplot(x='Quantity', y='Sales', data=data)
plt.title('Sales vs. Quantity')
plt.xlabel('Quantity')
plt.ylabel('Sales')
plt.show()
